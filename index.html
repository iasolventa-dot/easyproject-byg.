CREATE TABLE IF NOT EXISTS project_evaluations ( id SERIAL PRIMARY KEY, project_id INTEGER NOT NULL REFERENCES projects(id) ON DELETE CASCADE, department VARCHAR(50) NOT NULL CHECK (department IN ('tecnico','comercial','direccion')), decision VARCHAR(20) NOT NULL CHECK (decision IN ('pending','approved','rejected')), rejection_reason TEXT, evaluated_by VARCHAR(255), evaluated_at TIMESTAMPTZ, created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(), CONSTRAINT uq_project_eval UNIQUE (project_id, department) ); CREATE INDEX IF NOT EXISTS idx_project_evaluations_project ON project_evaluations(project_id); CREATE INDEX IF NOT EXISTS idx_project_evaluations_dept ON project_evaluations(department);
